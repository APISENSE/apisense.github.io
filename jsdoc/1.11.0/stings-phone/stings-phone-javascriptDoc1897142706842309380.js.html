 * Name of the network type, may be:
 * - GSM
 * - WCDMA
 * - LTE
 * @type {string}
 * @const
 */
Cell.prototype.networkType;
/**
 * Cell identifier, may be different depending on the network type.
 * - [GSM, WCDMA]: 16 bits identifier
 * - [LTE]: 28 bits identifier
 * @type {number}
 * @const
 */
Cell.prototype.cid;
/**
 * Identifier of the cell location.
 * - [GSM, WCDMA]: Location Area Code (LAC), 16 bits
 * - [LTE]: Tracking Area Code (TAC), 16 bits
 * @type {number}
 * @const
 */
Cell.prototype.lac;
/**
 * Mobile Country Code, 3 digits code.
 * @type {number}
 * @const
 */
Cell.prototype.mcc;
/**
 * Mobile Network Code, 2 or 3 digits code.
 * @type {number}
 * @const
 */
Cell.prototype.mnc;
/**
 * Received Signal Strength Indication, in DBm.
 * @type {number}
 * @const
 */
Cell.prototype.rssi;
/**
 * Received Signal Strength Indication, on a scale from 0 to 4.
 * This is the same level as shown on the phone UI.
 * @type {number}
 * @const
 */
Cell.prototype.signalLevel;
